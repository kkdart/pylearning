--Procedures
    CREATE OR REPLACE PROCEDURE return_nondiscounted_item(INT,INT)
    LANGUAGE plpgsql
    AS $$
    BEGIN
    UPDATE sales
    SET quantity = quantity -$2, order_total = order_total - price *  $2
    WHERE order_num =$1 AND discount = 0;
    COMMIT;
    END;
    $$;


RETURN a list of stored Procedures:
    SELECT
        routine_schema,
        routine_name
    FROM 
        information_schema.routines
    WHERE 
        routine_type = 'PROCEDURE';
